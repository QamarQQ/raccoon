# This input file defines a coupled PDE system:
# - (g(d) * u,x),x - b1 = 0
# - d,xx - u - b2 = 0
# g(d) = (1- d) ^ 2
# b1 = - 4 * x ^ 3 + 6 * x ^ 2 - 6 * x + 2
# b2 = - x + 2
# u(0) = 0, u(1) = 1
# v(0) = 0, v(1) = 0
# u = x, d = - x ^ 2 + x
# Each step the mesh is uniformly refined once.
# The manufactured solution should have a convergence order of 2.

[Mesh]
  type = GeneratedMesh
  dim = 2
  nx = 10
  ny = 1
[]

[Adaptivity]
  marker = 'box'
  [./Markers]
    [./box]
      type = BoxMarker
      bottom_left = '0 0 0'
      top_right = '1 1 0 '
      inside = 'refine'
      outside = 'do_nothing'
    [../]
  [../]
[]

  type = FixedPointProblem
[]

    type = ADFPIMaterialPropertyUserObject
    mat_prop = 'coupled_u'
